🚀 Quick Start

import { DatagridComponent, DataGridOptions } from '@darkoum/ng-datagrid';

@Component({
  standalone: true,
  imports: [DatagridComponent],
  template: `<app-datagrid [options]="gridOptions" />`
})
export class MyComponent {
  gridOptions: DataGridOptions = {
    keyExpr: 'id',
    dataSource: { store: myData },
    columns: [
      { dataField: 'name',      caption: 'ชื่อ' },
      { dataField: 'orderDate', caption: 'วันที่',   dataType: 'date', format: 'medium' },
      { dataField: 'total',     caption: 'ยอดรวม', format: '#,##0.00', alignment: 'right' },
    ],
    paging:  { enabled: true, pageSize: 20 },
    editing: { mode: 'popup', allowUpdating: true, allowDeleting: true },
    showBorders: true,
    rowAlternationEnabled: true,
  };
}

📊 DataGrid Options

ตัวเลือกหลัก

OptionTypeDefaultคำอธิบาย
keyExprstring—field ที่เป็น primary key (บังคับ)
dataSource.storeT[]—ข้อมูล array โดยตรง
dataSource.loadfunction—async load — รองรับ Observable
columnsGridColumn[]—คำจำกัดความของคอลัมน์ทั้งหมด
showBordersbooleanfalseแสดงเส้นขอบ grid
rowAlternationEnabledbooleanfalseสลับสีแถวคู่-คี่
hoverStateEnabledbooleanfalsehighlight แถวเมื่อ hover
thaiDatebooleanfalseแสดงวันที่เป็น พ.ศ. ทั้ง grid

GridColumn

PropertyTypeคำอธิบาย
dataFieldstringชื่อ field ใน data object
captionstringหัวคอลัมน์
widthnumberความกว้าง (px)
alignment'left'|'center'|'right'การจัดข้อความ
dataType'string'|'number'|'date'|'boolean'ประเภทข้อมูล
formatstring'#,##0.00', '0.00', 'medium', 'short' ฯลฯ
visiblebooleanแสดง/ซ่อน column (default: true)
allowEditingbooleanอนุญาตแก้ไข column นี้ (default: true)
lookup.dataSourceany[]ข้อมูลสำหรับ lookup (เหมือน dxo-lookup)
lookup.valueExprstringfield ที่เก็บค่า
lookup.displayExprstringfield ที่แสดงผล

✏️ Editing Modes

1. Row Editing — แก้ไข Inline

คลิก ✏️ แก้ไขตรงในแถว เหมาะกับข้อมูลที่ไม่ซับซ้อน

editing: {
  mode: 'row',
  allowUpdating: true,
  allowAdding: true,
  allowDeleting: true,
}

2. Popup Editing — Form ใน Dialog

เปิด popup เมื่อแก้ไข เหมาะกับ form หลาย field รองรับ group และ layout 2 คอลัมน์

editing: {
  mode: 'popup',
  allowUpdating: true,
  allowAdding: true,
  allowDeleting: true,
  popup: { title: 'แก้ไขข้อมูล', width: 800, height: 500 },
  form: {
    colCount: 2,
    items: [
      { dataField: 'name',   label: { text: 'ชื่อ' },       isRequired: true },
      { dataField: 'status', label: { text: 'สถานภาพ' }, editorType: 'dxSelectBox',
        editorOptions: { dataSource: statusList, valueExpr: 'id', displayExpr: 'name' } },
      { itemType: 'group', caption: 'ที่อยู่', colCount: 2, items: [
        { dataField: 'address', label: { text: 'ที่อยู่' } },
        { dataField: 'city',    label: { text: 'จังหวัด' } },
      ]},
    ]
  }
}

3. Tabbed Popup — Form หลายแท็บ (FormTab)

Popup ที่มีแท็บหลายแท็บ เหมาะกับ form ขนาดใหญ่ เช่น ข้อมูลนักศึกษา ลูกค้า

import { FormTab } from '@darkoum/ng-datagrid';

editing: {
  mode: 'popup',
  popup: { title: 'ข้อมูลนักศึกษา', width: 900, height: 600 },
  form: {
    tabs: [
      {
        title: '📋 ข้อมูลหลัก',
        items: [
          { dataField: 'studentcode', label: { text: 'รหัสนักศึกษา' }, isRequired: true },
          { dataField: 'name',        label: { text: 'ชื่อ' } },
        ]
      },
      {
        title: '🎓 การศึกษา',
        items: [
          { dataField: 'facultyid', label: { text: 'คณะ' }, editorType: 'dxSelectBox',
            editorOptions: { dataSource: facultyList, valueExpr: 'id', displayExpr: 'name' } },
          { dataField: 'admitdate', label: { text: 'วันที่รับเข้า' }, editorType: 'dxDateBox' },
        ]
      },
    ]
  }
}

      alert('กรุณากรอกชื่อ');
    }
  }
}

⚡ Events & Callbacks

EventPayloadเมื่อไร
onRowClick{ data, rowIndex, event }คลิกแถว
onRowDblClick{ data, rowIndex, event }ดับเบิลคลิกแถว
onRowRemoved{ data }หลังลบแถว
onSaving{ changes, cancel }ก่อนบันทึก (ตั้ง cancel=true เพื่อยกเลิก)
onSaved{ changes }หลังบันทึก
onDeleting{ data, cancel }ก่อนลบ
onDeleted{ data }หลังลบ
 
gridOptions: DataGridOptions = {
  // ...
  onRowClick: (e) => console.log('clicked:', e.data),
  onSaved:    (e) => this.reload(),
  onDeleted:  (e) => this.showToast('ลบสำเร็จ'),
}
